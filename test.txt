# ─────────────────────────────────────────────────
# 3. Send a message (non-streaming, message/send)
#    Should return a single JSON-RPC response
# ─────────────────────────────────────────────────

curl -s -X POST http://localhost:8080 \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer test-token-123" \
  -d '{
    "jsonrpc": "2.0",
    "id": "req-1",
    "method": "message/send",
    "params": {
      "message": {
        "kind": "message",
        "messageId": "msg-1",
        "role": "user",
        "parts": [
          { "kind": "text", "text": "What is 2 + 2?" }
        ]
      }
    }
  }' | jq .


# ─────────────────────────────────────────────────
# 4. Stream a message (SSE, message/stream)
#    -N disables curl buffering so you see events live
#    You should see JSON-RPC events arriving incrementally
# ─────────────────────────────────────────────────

curl -N -X POST http://localhost:8080 \
  -H "Content-Type: application/json" \
  -H "Accept: text/event-stream" \
  -H "Authorization: Bearer test-token-123" \
  -d '{
    "jsonrpc": "2.0",
    "id": "req-2",
    "method": "message/stream",
    "params": {
      "message": {
        "kind": "message",
        "messageId": "msg-2",
        "role": "user",
        "parts": [
          { "kind": "text", "text": "Explain what cloud run is in 3 sentences" }
        ]
      }
    }
  }'

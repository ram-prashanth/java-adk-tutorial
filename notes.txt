package com.enterprise.router.debug;

import jakarta.annotation.Priority;
import jakarta.ws.rs.container.ContainerRequestContext;
import jakarta.ws.rs.container.ContainerRequestFilter;
import jakarta.ws.rs.ext.Provider;
import org.jboss.logging.Logger;

import java.io.ByteArrayInputStream;
import java.io.IOException;
import java.nio.charset.StandardCharsets;

@Provider
@Priority(1) // Run first, before anything else
public class A2ARequestLoggingFilter implements ContainerRequestFilter {

    private static final Logger LOG = Logger.getLogger("a2a.debug");

    @Override
    public void filter(ContainerRequestContext ctx) throws IOException {
        // Log headers
        LOG.infof("═══ Incoming A2A Request ═══");
        LOG.infof("Method: %s %s", ctx.getMethod(), ctx.getUriInfo().getRequestUri());
        ctx.getHeaders().forEach((key, values) ->
            LOG.infof("Header: %s = %s", key, values)
        );

        // Read and log the full body
        byte[] body = ctx.getEntityStream().readAllBytes();
        String json = new String(body, StandardCharsets.UTF_8);
        LOG.infof("Body:\n%s", json);

        // CRITICAL: put the body back so the SDK can read it
        ctx.setEntityStream(new ByteArrayInputStream(body));
    }
}
